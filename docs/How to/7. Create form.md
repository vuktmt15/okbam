# Tạo Form

Base sẽ sử dụng react-hook-form cho việc quản lý state của form, về việc validate dữ liệu thì sẽ sử dụng yup

```tsx

import {Input} from "antd";

  getValidationSchema,
  ILoginForm,
} from "@app/module/login/NewPassword/form-config";  // Import interface và validation

interface ILoginForm {
  email?: string;
  password?: string;
}

export function LoginForm() {
  const loginMutation = useMutation(ApiUser.login);

  const handleLogin =
    ((

    ): void => {

      // ...
    },
    []);

  return (
    <FormGlobal<ILoginForm>
      resolver={getValidationSchema}
      onSubmit={handleLogin}
      render={({handleSubmitForm}) => {
        return (
          <div>
            <div className="header-wrapper">
              <Image
                className="login-image"
                src="img/logo.png"
                preview={false}
              />
              <div className="login-text">Đăng nhập</div>
            </div>
            <TextInputForm
              name="email"
              placeholder="Nhập tài khoản"

              name="password"

    />
  );
}
```
